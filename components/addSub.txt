function autoIcon(name) {
  if (!name) return null;
  const lower = name.toLowerCase();
  for (const key of Object.keys(SERVICE_ICONS)) {
    if (lower.includes(key)) return SERVICE_ICONS[key];
  }
  return null;
}

// ─── Renewal calc ─────────────────────────────────────────────────────────────
function calcRenewal(dateObj, cycle) {
  if (!dateObj) return null;
  const origD = dateObj.getDate();
  const today = new Date(); today.setHours(0,0,0,0);
  let next = new Date(dateObj); next.setHours(0,0,0,0);
  if (next > today) return next; // future date = that IS the next billing
  if (cycle === "Weekly") {
    while (next <= today) next.setDate(next.getDate() + 7);
  } else if (cycle === "Monthly") {
    while (next <= today) {
      let nm = next.getMonth() + 2, ny = next.getFullYear();
      if (nm > 12) { nm = 1; ny++; }
      const max = new Date(ny, nm, 0).getDate();
      next = new Date(ny, nm - 1, Math.min(origD, max));
    }
  } else if (cycle === "Yearly") {
    while (next <= today) next = new Date(next.getFullYear()+1, dateObj.getMonth(), origD);
  }
  return next;
}

function prettyDate(d) {
  if (!d) return "";
  return d.toLocaleDateString("en-GB", { day:"numeric", month:"short", year:"numeric" });
}

// ─── iOS-native date picker modal ─────────────────────────────────────────────
// Uses @react-native-community/datetimepicker which shows native spinner on iOS
function NativeDateModal({ visible, value, onConfirm, onCancel, theme }) {
  const [temp, setTemp] = useState(value || new Date());

  useEffect(() => { if (visible) setTemp(value || new Date()); }, [visible]);

  if (!visible) return null;

  // Dynamically require so Android doesn't crash if not installed
  let DateTimePicker = null;
  try { DateTimePicker = require("@react-native-community/datetimepicker").default; } catch {}

  if (!DateTimePicker) {
    // Fallback: simple text input
    return (
      <Modal visible transparent animationType="fade" onRequestClose={onCancel}>
        <View style={nd.overlay}>
          <View style={[nd.box, { backgroundColor: theme.card }]}>
            <Text style={[nd.title,{color:theme.text}]}>Enter date</Text>
            <TextInput
              style={[nd.textIn,{backgroundColor:theme.input,color:theme.text}]}
              placeholder="DD/MM/YYYY" placeholderTextColor={theme.subtext}
              onChangeText={v => {
                const [d,m,y] = v.split("/");
                if (y?.length===4) setTemp(new Date(+y,+m-1,+d));
              }}
            />
            <TouchableOpacity style={nd.doneBtn} onPress={()=>onConfirm(temp)}>
              <Text style={nd.doneTxt}>Done</Text>
            </TouchableOpacity>
          </View>
        </View>
      </Modal>
    );
  }

  if (Platform.OS === "android") {
    // Android shows inline native picker — just render it directly
    return (
      <DateTimePicker
        value={temp} mode="date" display="default"
        onChange={(e, d) => { if (d) onConfirm(d); else onCancel(); }}
      />
    );
  }

  // iOS — spinner in a modal sheet
  return (
    <Modal visible transparent animationType="slide" onRequestClose={onCancel}>
      <TouchableOpacity style={nd.overlay} activeOpacity={1} onPress={onCancel} />
      <View style={[nd.sheet, { backgroundColor: theme.card }]}>
        <View style={[nd.header, { borderBottomColor: theme.border }]}>
          <TouchableOpacity onPress={onCancel}>
            <Text style={[nd.cancelTxt,{color:theme.subtext}]}>Cancel</Text>
          </TouchableOpacity>
          <Text style={[nd.title,{color:theme.text}]}>First Billing Date</Text>
          <TouchableOpacity onPress={()=>onConfirm(temp)}>
            <Text style={nd.doneTxt}>Done</Text>
          </TouchableOpacity>
        </View>
        <DateTimePicker
          value={temp} mode="date" display="spinner"
          textColor={theme.text}
          onChange={(_, d) => { if (d) setTemp(d); }}
          style={{ width:"100%" }}
        />
      </View>
    </Modal>
  );
}

// ─── Icon picker modal (works on iOS & Android) ────────────────────────────────
function IconPickerModal({ visible, current, onSelect, onClose, theme }) {
  const { height: SH } = Dimensions.get("window");
  const translateY = useRef(new Animated.Value(SH)).current;

  useEffect(() => {
    Animated.spring(translateY, {
      toValue: visible ? 0 : SH, useNativeDriver: true, damping: 22, stiffness: 220,
    }).start();
  }, [visible]);

  const pan = useRef(PanResponder.create({
    onStartShouldSetPanResponder: () => true,
    onMoveShouldSetPanResponder: (_, gs) => gs.dy > 4,
    onPanResponderMove: (_, gs) => { if (gs.dy > 0) translateY.setValue(gs.dy); },
    onPanResponderRelease: (_, gs) => {
      if (gs.dy > 80 || gs.vy > 0.5) {
        Animated.timing(translateY,{toValue:SH,duration:220,useNativeDriver:true}).start(onClose);
      } else {
        Animated.spring(translateY,{toValue:0,useNativeDriver:true,damping:22,stiffness:220}).start();
      }
    },
  })).current;

  const pickFromGallery = async () => {
    const perm = await ImagePicker.requestMediaLibraryPermissionsAsync();
    if (!perm.granted) { Alert.alert("Permission needed","Allow photo access to use gallery images."); return; }
    const result = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.Images,
      allowsEditing: true, aspect:[1,1], quality:0.8,
    });
    if (!result.canceled && result.assets?.[0]?.uri) {
      onSelect({ type:"image", uri: result.assets[0].uri });
      onClose();
    }
  };

  if (!visible) return null;

  return (
    <Modal visible transparent animationType="none" onRequestClose={onClose}>
      <TouchableOpacity style={ip.backdrop} activeOpacity={1} onPress={onClose} />
      <Animated.View style={[ip.sheet, { backgroundColor: theme.card, transform:[{translateY}] }]}>
        <View {...pan.panHandlers} style={ip.handleArea}>
          <View style={[ip.handle,{backgroundColor:theme.border}]} />
        </View>
        <Text style={[ip.title,{color:theme.text}]}>Choose Icon</Text>

        {/* Gallery option */}
        <TouchableOpacity style={[ip.galleryBtn,{backgroundColor:theme.input}]} onPress={pickFromGallery}>
          <Ionicons name="image-outline" size={22} color="#8e44ad" />
          <Text style={[ip.galleryTxt,{color:theme.text}]}>Choose from Gallery</Text>
          <Ionicons name="chevron-forward" size={16} color={theme.subtext} />
        </TouchableOpacity>

        <Text style={[ip.sectionLabel,{color:theme.subtext}]}>Or pick an emoji</Text>

        <ScrollView showsVerticalScrollIndicator={false} contentContainerStyle={ip.grid}>
          {EMOJI_ICONS.map(ic => (
            <TouchableOpacity
              key={ic} onPress={() => { onSelect({ type:"emoji", value:ic }); onClose(); }}
              style={[ip.item, {backgroundColor:theme.input},
                current?.type==="emoji"&&current.value===ic && ip.itemSelected]}
            >
              <Text style={ip.emoji}>{ic}</Text>
            </TouchableOpacity>
          ))}
        </ScrollView>
      </Animated.View>
    </Modal>
  );
}

  sheet: { position:"absolute", bottom:0, left:0, right:0, borderTopLeftRadius:24,

// ─── Row + Group helpers (clean grouped card layout) ─────────────────────────
function RowItem({ label, right, theme, noBorder }) {
  return (
    <View style={[rS.row, {borderBottomColor:theme.border}, noBorder&&{borderBottomWidth:0}]}>
      <Text style={[rS.label,{color:theme.text}]}>{label}</Text>
      <View style={rS.right}>{right}</View>
    </View>
  );
}
const rS = StyleSheet.create({
  row: { flexDirection:"row", justifyContent:"space-between", alignItems:"center",
    paddingVertical:14, borderBottomWidth:StyleSheet.hairlineWidth },
  label: { fontSize:16, fontWeight:"500" },
  right: { flexDirection:"row", alignItems:"center", gap:6 },
});

function Group({ children, theme }) {
  return (
    <View style={[gS.card, {backgroundColor:theme.card, borderColor:theme.border}]}>
      {children}
    </View>
  );
}
const gS = StyleSheet.create({
  card: { borderRadius:16, paddingHorizontal:16, marginBottom:12,
    borderWidth:StyleSheet.hairlineWidth },
});

// ─── Generic bottom-sheet picker modal (iOS & Android) ───────────────────────
function PickerModal({ visible, title, options, current, onSelect, onClose, theme }) {
  return (
    <Modal visible={visible} transparent animationType="slide" onRequestClose={onClose}>
      <TouchableOpacity style={pm.backdrop} activeOpacity={1} onPress={onClose} />
      <View style={[pm.sheet, {backgroundColor:theme.card}]}>
        <View style={[pm.header,{borderBottomColor:theme.border}]}>
          <Text style={[pm.title,{color:theme.text}]}>{title}</Text>
          <TouchableOpacity onPress={onClose}>
            <Text style={pm.done}>Done</Text>
          </TouchableOpacity>
        </View>
        <ScrollView style={pm.list}>
          {options.map((opt,i) => (
            <TouchableOpacity
              key={opt}
              style={[pm.item, i<options.length-1&&{borderBottomWidth:StyleSheet.hairlineWidth,borderBottomColor:theme.border}]}
              onPress={() => { onSelect(opt); onClose(); }}
            >
              <Text style={[pm.itemText,{color:theme.text}, current===opt&&{color:"#8e44ad",fontWeight:"700"}]}>{opt}</Text>
              {current===opt && <Ionicons name="checkmark" size={18} color="#8e44ad" />}
            </TouchableOpacity>
          ))}
        </ScrollView>
      </View>
    </Modal>
  );
}
